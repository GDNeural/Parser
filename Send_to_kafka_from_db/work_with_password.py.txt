import base64


def encodeText(text):
    ascii_text = text.encode('ascii')
    encoded_to_bytes = base64.b64encode(ascii_text)
    encoded_text = encoded_to_bytes.decode('ascii')
    return encoded_text


def decodeText(encrypted_text):
    ascii_text = encrypted_text.encode('ascii')
    encoded_to_bytes = base64.b64encode(ascii_text)
    decoded_text = encoded_to_bytes.decode('ascii')
    return decoded_text
